function cloneDeep(target, hash = new WeakMap()) {
  if (target === null || typeof target !== 'object') {
    return target; // åŸå§‹ç±»å‹ç›´æ¥è¿”å›

  if (hash.has(target)) {
    return hash.get(target); // é¿å…å¾ªç¯å¼•ç”¨

      const value = target[key];

  }

  return result;
