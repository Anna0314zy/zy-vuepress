еүҚиЁҖ

еҶҷе®ҢдёҠдёҖзҜҮж–Үз« жғіеӯҰNode.jsпјҢstreamе…Ҳжңүеҝ…иҰҒжҗһжё…жҘҡ

BufferжҺўз©¶

зңӢдёҖж®өд№ӢеүҚдҪҝз”Ёstreamж“ҚдҪңж–Үд»¶зҡ„дҫӢеӯҗпјҡ

console.log('streamеҶ…е®№',stream);  
stream.on('data',function(chunk){

BufferеҜ№иұЎпјҢзұ»дјјж•°з»„пјҢе®ғзҡ„е…ғзҙ дёә16иҝӣеҲ¶зҡ„дёӨдҪҚж•°пјҢеҚі0еҲ°255зҡ„ж•°еҖјгҖӮеҸҜд»ҘзңӢеҮәstreamдёӯжөҒеҠЁзҡ„ж•°жҚ®жҳҜBufferзұ»еһӢпјҢдәҢиҝӣеҲ¶ж•°жҚ®пјҢжҺҘдёӢжқҘејҖе§ӢжҲ‘д»¬зҡ„BufferжҺўзҙўд№Ӣж—…гҖӮ

д»Җд№ҲжҳҜдәҢиҝӣеҲ¶

дәҢиҝӣеҲ¶жҳҜи®Ўз®—жңәжңҖеә•еұӮзҡ„ж•°жҚ®ж јејҸпјҢеӯ—з¬ҰдёІпјҢж•°еӯ—пјҢи§Ҷйў‘пјҢйҹійў‘пјҢзЁӢеәҸпјҢзҪ‘з»ңеҢ…зӯүпјҢеңЁжңҖеә•еұӮйғҪжҳҜз”ЁдәҢиҝӣеҲ¶жқҘиҝӣиЎҢеӯҳеӮЁгҖӮиҝҷдәӣй«ҳзә§ж јејҸе’ҢдәҢиҝӣеҲ¶д№Ӣй—ҙпјҢйғҪеҸҜд»ҘйҖҡиҝҮеӣәе®ҡзҡ„зј–з Ғж јејҸиҝӣиЎҢзӣёдә’иҪ¬жҚўгҖӮ

дҫӢеҰӮпјҢCиҜӯиЁҖдёӯint32зұ»еһӢзҡ„еҚҒиҝӣеҲ¶ж•ҙж•°пјҲж— з¬ҰеҸ·пјүпјҢе°ұеҚ з”Ё32bitеҚі4byteпјҢеҚҒиҝӣеҲ¶зҡ„3еҜ№еә”зҡ„дәҢиҝӣеҲ¶е°ұжҳҜ00000000 00000000 00000000 00000011гҖӮеӯ—з¬ҰдёІд№ҹжҳҜеҗҢзҗҶпјҢеҸҜд»Ҙж №жҚ®ASCIIзј–з Ғи§„еҲҷжҲ–иҖ…unicodeзј–з Ғи§„еҲҷпјҲеҰӮutf-8пјүзӯүе’ҢдәҢиҝӣеҲ¶иҝӣиЎҢзӣёдә’иҪ¬жҚўгҖӮжҖ»д№ӢпјҢи®Ўз®—жңәеә•еұӮеӯҳеӮЁзҡ„ж•°жҚ®йғҪжҳҜдәҢиҝӣеҲ¶ж јејҸпјҢеҗ„з§Қй«ҳзә§зұ»еһӢйғҪжңүеҜ№еә”зҡ„зј–з Ғи§„еҲҷе’ҢдәҢиҝӣеҲ¶иҝӣиЎҢзӣёдә’иҪ¬жҚўгҖӮ

nodeдёӯдёәд»Җд№ҲдјҡеҮәзҺ°BufferиҝҷдёӘжЁЎеқ—

еңЁжңҖеҲқзҡ„javascriptз”ҹжҖҒдёӯпјҢjavascriptиҝҳиҝҗиЎҢеңЁжөҸи§ҲеҷЁз«ҜпјҢеҜ№дәҺеӨ„зҗҶUnicodeзј–з Ғзҡ„еӯ—з¬ҰдёІж•°жҚ®еҫҲе®№жҳ“пјҢдҪҶжҳҜеҜ№дәҺеӨ„зҗҶдәҢиҝӣеҲ¶д»ҘеҸҠйқһUnicodeзј–з Ғзҡ„ж•°жҚ®ж— иғҪдёәеҠӣпјҢдҪҶжҳҜеҜ№дәҺServerз«Ҝж“ҚдҪңTCP/HTTPд»ҘеҸҠж–Үд»¶I/Oзҡ„еӨ„зҗҶжҳҜеҝ…йЎ»зҡ„гҖӮжҲ‘жғіе°ұжҳҜеӣ жӯӨеңЁNode.jsйҮҢйқўжҸҗдҫӣдәҶBufferзұ»еӨ„зҗҶдәҢиҝӣеҲ¶зҡ„ж•°жҚ®пјҢеҸҜд»ҘеӨ„зҗҶеҗ„з§Қзұ»еһӢзҡ„ж•°жҚ®гҖӮ

BufferжЁЎеқ—зҡ„дёҖдёӘиҜҙжҳҺгҖӮ

еңЁNode.jsйҮҢйқўдёҖдәӣйҮҚиҰҒжЁЎеқ—netгҖҒhttpгҖҒfsдёӯзҡ„ж•°жҚ®дј иҫ“д»ҘеҸҠеӨ„зҗҶйғҪжңүBufferзҡ„иә«еҪұпјҢеӣ дёәдёҖдәӣеҹәзЎҖзҡ„ж ёеҝғжЁЎеқ—йғҪиҰҒдҫқиө–BufferпјҢжүҖд»ҘеңЁnodeеҗҜеҠЁзҡ„ж—¶еҖҷпјҢе°ұе·Із»ҸеҠ иҪҪдәҶBufferпјҢжҲ‘д»¬еҸҜд»ҘеңЁе…ЁеұҖдёӢйқўзӣҙжҺҘдҪҝз”ЁBufferпјҢж— йңҖйҖҡиҝҮrequire()гҖӮдё” Buffer зҡ„еӨ§е°ҸеңЁеҲӣе»әж—¶зЎ®е®ҡпјҢж— жі•и°ғж•ҙгҖӮ

BufferеҲӣе»ә

еңЁ NodeJS v6.0.0зүҲжң¬д№ӢеүҚпјҢBufferе®һдҫӢжҳҜйҖҡиҝҮ Buffer жһ„йҖ еҮҪж•°еҲӣе»әзҡ„пјҢеҚідҪҝз”Ё new е…ій”®еӯ—еҲӣе»әпјҢе®ғж №жҚ®жҸҗдҫӣзҡ„еҸӮж•°иҝ”еӣһдёҚеҗҢзҡ„ BufferпјҢдҪҶеңЁд№ӢеҗҺзҡ„зүҲжң¬дёӯиҝҷз§ҚеЈ°жҳҺж–№ејҸе°ұиў«еәҹејғдәҶпјҢжӣҝд»Ј new зҡ„еҲӣе»әж–№ејҸдё»иҰҒжңүд»ҘдёӢеҮ з§ҚгҖӮ

1. Buffer.alloc е’Ң Buffer.allocUnsafe(еҲӣе»әеӣәе®ҡеӨ§е°Ҹзҡ„buffer)

з”Ё Buffer.alloc е’Ң Buffer.allocUnsafe еҲӣе»ә Buffer зҡ„дј еҸӮж–№ејҸзӣёеҗҢпјҢеҸӮж•°дёәеҲӣе»ә Buffer зҡ„й•ҝеәҰпјҢж•°еҖјзұ»еһӢгҖӮ

// Buffer.alloc е’Ң Buffer.allocUnsafe еҲӣе»ә Buffer
// Buffer.alloc еҲӣе»ә Buffer,еҲӣе»әдёҖдёӘеӨ§е°Ҹдёә6еӯ—иҠӮзҡ„з©әbufferпјҢз»ҸиҝҮдәҶеҲқе§ӢеҢ–
let buf1 = Buffer.alloc(6);

// Buffer.allocUnsafe еҲӣе»ә BufferпјҢеҲӣе»әдёҖдёӘеӨ§е°Ҹдёә6еӯ—иҠӮзҡ„bufferпјҢжңӘз»ҸиҝҮеҲқе§ӢеҢ–
let buf2 = Buffer.allocUnsafe(6);

console.log(buf1); // <Buffer 00 00 00 00 00 00>
console.log(buf2); // <Buffer 00 e7 8f a0 00 00>

йҖҡиҝҮд»Јз ҒеҸҜд»ҘзңӢеҮәпјҢз”Ё Buffer.alloc е’Ң Buffer.allocUnsafe еҲӣе»ә Buffer жҳҜжңүеҢәеҲ«зҡ„пјҢBuffer.alloc еҲӣе»әзҡ„ Buffer жҳҜиў«еҲқе§ӢеҢ–иҝҮзҡ„пјҢеҚі Buffer зҡ„жҜҸдёҖйЎ№йғҪз”Ё 00 еЎ«е……пјҢиҖҢ Buffer.allocUnsafe еҲӣе»әзҡ„ Buffer е№¶жІЎжңүз»ҸиҝҮеҲқе§ӢеҢ–пјҢеңЁеҶ…еӯҳдёӯеҸӘиҰҒжңүй—ІзҪ®зҡ„ Buffer е°ұзӣҙжҺҘ вҖңжҠ“иҝҮжқҘвҖқ дҪҝз”ЁгҖӮ

Buffer.allocUnsafe еҲӣе»ә Buffer дҪҝеҫ—еҶ…еӯҳзҡ„еҲҶй…Қйқһеёёеҝ«пјҢдҪҶе·ІеҲҶй…Қзҡ„еҶ…еӯҳж®өеҸҜиғҪеҢ…еҗ«жҪңеңЁзҡ„ж•Ҹж„ҹж•°жҚ®пјҢжңүжҳҺжҳҫжҖ§иғҪдјҳеҠҝзҡ„еҗҢж—¶еҸҲжҳҜдёҚе®үе…Ёзҡ„пјҢжүҖд»ҘдҪҝз”ЁйңҖж јеӨ– вҖңе°ҸеҝғвҖқгҖӮ

2гҖҒBuffer.from(ж №жҚ®еҶ…е®№зӣҙжҺҘеҲӣе»әBuffer)

Buffer.from(str, )
ж”ҜжҢҒдёүз§Қдј еҸӮж–№ејҸпјҡ

  • з¬¬дёҖдёӘеҸӮж•°дёәеӯ—з¬ҰдёІпјҢз¬¬дәҢдёӘеҸӮж•°дёәеӯ—з¬Ұзј–з ҒпјҢеҰӮ ASCIIгҖҒUTF-8гҖҒBase64 зӯүзӯүгҖӮ
  • дј е…ҘдёҖдёӘж•°з»„пјҢж•°з»„зҡ„жҜҸдёҖйЎ№дјҡд»ҘеҚҒе…ӯиҝӣеҲ¶еӯҳеӮЁдёә Buffer зҡ„жҜҸдёҖйЎ№гҖӮ
  • дј е…ҘдёҖдёӘ BufferпјҢдјҡе°Ҷ Buffer зҡ„жҜҸдёҖйЎ№дҪңдёәж–°иҝ”еӣһ Buffer зҡ„жҜҸдёҖйЎ№гҖӮ

иҜҙжҳҺ:Bufferзӣ®еүҚж”ҜжҢҒзҡ„зј–з Ғж јејҸ

  • ascii - д»…ж”ҜжҢҒ7дҪҚASCIIж•°жҚ®гҖӮ
  • utf8 - еӨҡеӯ—иҠӮзј–з Ғзҡ„Unicodeеӯ—з¬Ұ
  • utf16le - 2жҲ–4дёӘеӯ—иҠӮпјҢе°Ҹз«Ҝзј–з Ғзҡ„Unicodeеӯ—з¬Ұ
  • base64 - Base64еӯ—з¬ҰдёІзј–з Ғ
  • binary - дәҢиҝӣеҲ¶зј–з ҒгҖӮ
  • hex - е°ҶжҜҸдёӘеӯ—иҠӮзј–з ҒдёәдёӨдёӘеҚҒе…ӯиҝӣеҲ¶еӯ—з¬ҰгҖӮ
дј е…Ҙеӯ—з¬ҰдёІе’Ңеӯ—з¬Ұзј–з Ғпјҡ
// дј е…Ҙеӯ—з¬ҰдёІе’Ңеӯ—з¬Ұзј–з Ғ

console.log(buf); // <Buffer 68 65 6c 6c 6f>
дј е…Ҙж•°з»„пјҡ
// ж•°з»„жҲҗе‘ҳдёәеҚҒиҝӣеҲ¶ж•°

console.log(buf); // <Buffer 01 02 03>
// ж•°з»„жҲҗе‘ҳдёәеҚҒе…ӯиҝӣеҲ¶ж•°
let buf = Buffer.from([0xe4, 0xbd, 0xa0, 0xe5, 0xa5, 0xbd]);

console.log(buf); // <Buffer e4 bd a0 e5 a5 bd>
console.log(buf.toString("utf8")); // дҪ еҘҪ

еңЁ NodeJS дёӯдёҚж”ҜжҢҒ GB2312 зј–з ҒпјҢй»ҳи®Өж”ҜжҢҒ UTF-8пјҢеңЁ GB2312 дёӯпјҢдёҖдёӘжұүеӯ—еҚ дёӨдёӘеӯ—иҠӮпјҢиҖҢеңЁ UTF-8 дёӯпјҢдёҖдёӘжұүеӯ—еҚ дёүдёӘеӯ—иҠӮпјҢжүҖд»ҘдёҠйқў вҖңдҪ еҘҪвҖқ зҡ„ Buffer дёә 6 дёӘеҚҒе…ӯиҝӣеҲ¶ж•°з»„жҲҗгҖӮ

// ж•°з»„жҲҗе‘ҳдёәеӯ—з¬ҰдёІзұ»еһӢзҡ„ж•°еӯ—
let buf = Buffer.from(["1", "2", "3"]);
console.log(buf); // <Buffer 01 02 03>

дј е…Ҙзҡ„ж•°з»„жҲҗе‘ҳеҸҜд»ҘжҳҜд»»дҪ•иҝӣеҲ¶зҡ„ж•°еҖјпјҢеҪ“жҲҗе‘ҳдёәеӯ—з¬ҰдёІзҡ„ж—¶еҖҷпјҢеҰӮжһңеҖјжҳҜж•°еӯ—дјҡиў«иҮӘеҠЁиҜҶеҲ«жҲҗж•°еҖјзұ»еһӢпјҢеҰӮжһңеҖјдёҚжҳҜж•°еӯ—жҲ–жҲҗе‘ҳдёәжҳҜе…¶д»–йқһж•°еҖјзұ»еһӢзҡ„ж•°жҚ®пјҢиҜҘжҲҗе‘ҳдјҡиў«еҲқе§ӢеҢ–дёә 00гҖӮ

еҲӣе»әзҡ„ Buffer еҸҜд»ҘйҖҡиҝҮ toString ж–№жі•зӣҙжҺҘжҢҮе®ҡзј–з ҒиҝӣиЎҢиҪ¬жҚўпјҢй»ҳи®Өзј–з Ғдёә UTF-8гҖӮ

дј е…Ҙ Bufferпјҡ
// дј е…ҘдёҖдёӘ Buffer
let buf1 = Buffer.from("hello", "utf8");

let buf2 = Buffer.from(buf1);

console.log(buf1); // <Buffer 68 65 6c 6c 6f>
console.log(buf2); // <Buffer 68 65 6c 6c 6f>
console.log(buf1 === buf2); // false
console.log(buf1[0] === buf2[0]); // true
buf1[1]=12;
console.log(buf1); // <Buffer 68 0c 6c 6c 6f>
console.log(buf2); // <Buffer 68 65 6c 6c 6f>

еҪ“дј е…Ҙзҡ„еҸӮж•°дёәдёҖдёӘ Buffer зҡ„ж—¶еҖҷпјҢдјҡеҲӣе»әдёҖдёӘж–°зҡ„ Buffer е№¶еӨҚеҲ¶дёҠйқўзҡ„жҜҸдёҖдёӘжҲҗе‘ҳгҖӮ

Buffer дёәеј•з”Ёзұ»еһӢпјҢдёҖдёӘ Buffer еӨҚеҲ¶дәҶеҸҰдёҖдёӘ Buffer зҡ„жҲҗе‘ҳпјҢеҪ“е…¶дёӯдёҖдёӘ Buffer еӨҚеҲ¶зҡ„жҲҗе‘ҳжңүжӣҙж”№пјҢеҸҰдёҖдёӘ Buffer еҜ№еә”зҡ„жҲҗе‘ҳдёҚдјҡи·ҹзқҖж”№еҸҳпјҢиҜҙжҳҺдј е…ҘbufferеҲӣе»әж–°зҡ„Bufferзҡ„ж—¶еҖҷжҳҜдёҖдёӘж·ұжӢ·иҙқзҡ„иҝҮзЁӢгҖӮ

Buffer.poolSize = 8 * 1024;

function allocate(size)
{
    if(size <= 0 )
        return new FastBuffer();
    if(size < Buffer.poolSize >>> 1 )
        if(size > poolSize - poolOffset)
            createPool();
        var b = allocPool.slice(poolOffset,poolOffset + size);
        poolOffset += size;
        alignPool();
        return b
    } else {
        return createUnsafeBuffer(size);
    }
}

  • еҲҶй…ҚеӨ§еҶ…еӯҳ

  • еҶ…еӯҳеҲҶй…Қзҡ„йҷҗеҲ¶

жӯӨеӨ–пјҢBufferеҚ•ж¬Ўзҡ„еҶ…еӯҳеҲҶй…Қд№ҹжңүйҷҗеҲ¶пјҢиҖҢиҝҷдёӘйҷҗеҲ¶ж №жҚ®дёҚеҗҢж“ҚдҪңзі»з»ҹиҖҢдёҚеҗҢпјҢиҖҢиҝҷдёӘйҷҗеҲ¶еҸҜд»ҘзңӢеҲ°node_buffer.hйҮҢйқў

    sizeof(int32_t) == sizeof(intptr_t) ? 0x3fffffff : 0x7fffffff;

bufferеҶ…еӯҳеҲҶй…ҚжңәеҲ¶дјҳзӮ№

BufferзңҹжӯЈзҡ„еҶ…еӯҳе®һеңЁNodeзҡ„C++еұӮйқўжҸҗдҫӣзҡ„пјҢJavaScriptеұӮйқўеҸӘжҳҜдҪҝз”Ёе®ғгҖӮеҪ“иҝӣиЎҢе°ҸиҖҢйў‘з№Ғзҡ„Bufferж“ҚдҪңж—¶пјҢйҮҮз”Ёзҡ„жҳҜ8KBдёәдёҖдёӘеҚ•е…ғзҡ„жңәеҲ¶иҝӣиЎҢйў„е…Ҳз”іиҜ·е’ҢдәӢеҗҺеҲҶй…ҚпјҢдҪҝеҫ—JavascriptеҲ°ж“ҚдҪңзі»з»ҹд№Ӣй—ҙдёҚеҝ…жңүиҝҮеӨҡзҡ„еҶ…еӯҳз”іиҜ·ж–№йқўзҡ„зі»з»ҹи°ғз”ЁгҖӮеҜ№дәҺеӨ§еқ—зҡ„BufferиҖҢиЁҖ(еӨ§дәҺ8KB)пјҢеҲҷзӣҙжҺҘдҪҝз”ЁC++еұӮйқўжҸҗдҫӣзҡ„еҶ…еӯҳпјҢеҲҷж— йңҖз»Ҷи…»зҡ„еҲҶй…Қж“ҚдҪңгҖӮ

BufferдёҺstream

еҺҹеӣ дёҖпјҡ

nodeе®ҳж–№дҪҝз”ЁдәҢиҝӣеҲ¶дҪңдёәж•°жҚ®жөҒеҠЁиӮҜе®ҡжҳҜиҖғиҷ‘иҝҮеҫҲеӨҡпјҢжҜ”еҰӮеңЁдёҠдёҖзҜҮ жғіеӯҰNode.jsпјҢstreamе…Ҳжңүеҝ…иҰҒжҗһжё…жҘҡж–Үз« е·Із»ҸиҜҙиҝҮпјҢstreamдё»иҰҒзҡ„и®ҫи®Ўзӣ®зҡ„вҖ”вҖ”жҳҜдёәдәҶдјҳеҢ–IOж“ҚдҪңпјҲж–Үд»¶IOе’ҢзҪ‘з»ңIOпјүпјҢеҜ№еә”еҗҺз«Ҝж— и®әжҳҜж–Үд»¶IOиҝҳжҳҜзҪ‘з»ңIOпјҢе…¶дёӯеҢ…еҗ«зҡ„ж•°жҚ®ж јејҸйғҪжҳҜжңӘзҹҘзҡ„пјҢжңүеҸҜиғҪжҳҜеӯ—з¬ҰдёІпјҢйҹійў‘пјҢи§Ҷйў‘пјҢзҪ‘з»ңеҢ…зӯүзӯүпјҢеҚідҪҝе°ұжҳҜеӯ—з¬ҰдёІпјҢе®ғзҡ„зј–з Ғж јејҸд№ҹжҳҜжңӘзҹҘзҡ„пјҢеҸҜиғҪASCзј–з ҒпјҢд№ҹеҸҜиғҪutf-8зј–з ҒпјҢеҜ№дәҺиҝҷдәӣжңӘзҹҘзҡ„жғ…еҶөпјҢиҝҳдёҚеҰӮзӣҙжҺҘдҪҝз”ЁжңҖйҖҡз”Ёзҡ„ж јејҸдәҢиҝӣеҲ¶.

е°Ҷд»Јз Ғдёӯзҡ„buffer-testж–Үд»¶еӨ§е°ҸеўһеҠ еҲ°50KBе·ҰеҸіпјҢз„¶еҗҺдҪҝз”Ёabе·Ҙе…·жөӢиҜ•дёҖдёӢжҖ§иғҪпјҢдҪ дјҡеҸ‘зҺ°ж— и®әжҳҜд»ҺеҗһеҗҗйҮҸпјҲRequests per secondпјүиҝҳжҳҜиҝһжҺҘж—¶й—ҙдёҠпјҢиҝ”еӣһдәҢиҝӣеҲ¶ж јејҸжҜ”иҝ”еӣһеӯ—з¬ҰдёІж јејҸж•ҲзҺҮжҸҗй«ҳеҫҲеӨҡгҖӮдёәдҪ•еӯ—з¬ҰдёІж јејҸж•ҲзҺҮдҪҺпјҹвҖ”вҖ” еӣ дёәзҪ‘з»ңиҜ·жұӮзҡ„ж•°жҚ®жң¬жқҘе°ұжҳҜдәҢиҝӣеҲ¶ж јејҸдј иҫ“пјҢиҷҪз„¶д»Јз ҒдёӯеҶҷзҡ„жҳҜ response иҝ”еӣһеӯ—з¬ҰдёІпјҢжңҖз»Ҳиҝҳеҫ—еҶҚиҪ¬жҚўдёәдәҢиҝӣеҲ¶иҝӣиЎҢдј иҫ“пјҢеӨҡдәҶдёҖжӯҘж“ҚдҪңпјҢж•ҲзҺҮеҪ“з„¶дҪҺдәҶгҖӮ

BufferеңЁstreamж•°жҚ®жөҒиҪ¬е……еҪ“зҡ„и§’иүІ

жҲ‘д»¬еҸҜд»ҘжҠҠж•ҙдёӘжөҒ(stream)е’ҢBufferзҡ„й…ҚеҗҲиҝҮзЁӢзңӢдҪңе…¬дәӨз«ҷгҖӮеңЁдёҖдәӣе…¬дәӨз«ҷпјҢе…¬дәӨиҪҰеңЁжІЎжңүиЈ…ж»Ўд№ҳе®ўеүҚжҳҜдёҚдјҡеҸ‘иҪҰзҡ„пјҢжҲ–иҖ…еңЁзү№е®ҡзҡ„ж—¶еҲ»жүҚдјҡеҸ‘иҪҰгҖӮеҪ“з„¶пјҢд№ҳе®ўд№ҹеҸҜиғҪеңЁдёҚеҗҢзҡ„ж—¶й—ҙпјҢдәәжөҒйҮҸеӨ§е°Ҹд№ҹдјҡжңүжүҖдёҚеҗҢпјҢжңүдәәеӨҡзҡ„ж—¶еҖҷпјҢжңүдәәе°‘зҡ„ж—¶еҖҷпјҢд№ҳе®ўжҲ–е…¬дәӨиҪҰз«ҷйғҪж— жі•жҺ§еҲ¶дәәжөҒйҮҸгҖӮ

дёҚи®әдҪ•ж—¶пјҢж—©еҲ°зҡ„д№ҳе®ўйғҪеҝ…йЎ»зӯүеҫ…пјҢзӣҙеҲ°е…¬дәӨиҪҰжҺҘеҲ°жҢҮд»ӨеҸҜд»ҘеҸ‘иҪҰгҖӮеҪ“д№ҳе®ўеҲ°з«ҷпјҢеҸ‘зҺ°е…¬дәӨиҪҰе·Із»ҸиЈ…ж»ЎпјҢжҲ–иҖ…е·Із»ҸејҖиө°пјҢд»–е°ұеҝ…йЎ»зӯүеҫ…дёӢдёҖзҸӯиҪҰж¬ЎгҖӮ

жҖ»д№ӢпјҢиҝҷйҮҢжҖ»дјҡжңүдёҖдёӘзӯүеҫ…зҡ„ең°ж–№пјҢиҝҷдёӘзӯүеҫ…зҡ„еҢәеҹҹе°ұжҳҜNode.jsдёӯзҡ„BufferпјҢNode.jsдёҚиғҪжҺ§еҲ¶ж•°жҚ®д»Җд№Ҳж—¶еҖҷдј иҫ“иҝҮжқҘпјҢдј иҫ“йҖҹеәҰпјҢе°ұеҘҪеғҸе…¬дәӨиҪҰз«ҷж— жі•жҺ§еҲ¶дәәжөҒйҮҸдёҖж ·гҖӮд»–еҸӘиғҪеҶіе®ҡд»Җд№Ҳж—¶еҖҷеҸ‘йҖҒж•°жҚ®(е…¬дәӨиҪҰеҸ‘иҪҰ)гҖӮеҰӮжһңж—¶й—ҙиҝҳдёҚеҲ°пјҢйӮЈд№ҲNode.jsе°ұдјҡжҠҠж•°жҚ®ж”ҫе…ҘBufferзӯүеҫ…еҢәеҹҹдёӯпјҢдёҖдёӘеңЁRAMдёӯзҡ„ең°еқҖпјҢзӣҙеҲ°жҠҠд»–д»¬еҸ‘йҖҒеҮәеҺ»иҝӣиЎҢеӨ„зҗҶгҖӮ
